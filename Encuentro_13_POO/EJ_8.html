21         System.out.println("Por favor, ingrese el tamanio del cuadrado. Tiene que ser un numero par");
22         Scanner read = new Scanner(System.in);
23         int size = read.nextInt();
24 
25         if (size % 2 != 0) {
26             do {
27                 System.out.println("Por favor ingrese un tamanio par");
28                 size = read.nextInt();
29             } while (size % 2 != 0);
30         }
31 
32         //Hasta ahora lo que se hace es ingresar el size del cuadrado
33         System.out.println("El cuadrado de tamanio " + size + " es");
34         for (int i = 0; i < size; i++) {
35             for (int j = 0; j < size; j++) {
36                 if (i > 0 && i < size - 1 && j > 0 && j < size - 1) {
37                     System.out.print("   ");
38                 } else {
39                     System.out.print(" * ");
40                 }
41 
42             }
43             System.out.println("");
44         }
45 
46     }
